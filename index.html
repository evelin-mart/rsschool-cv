// generation of unique item sequency

function generateCardSequence(perPage, pagesCount, itemsCount) {
  const sequency = [];
  for (let i = 0; i < pagesCount; i++) {
    const miniSequence = new Set();
    while (miniSequence.size < perPage) {
      const num = getRandomInt(0, itemsCount - 1);
      const count = sequency
        .filter(item => item === num).length;
      const maxCount = Math.floor((sequency.length + miniSequence.size) / itemsCount);
      if (count <= maxCount) {
        miniSequence.add(num)
      }
    }
    sequency.push(...miniSequence)
  }
  return sequency
}
